2)use SourceSystem;
Select Product.ProductID, ProductCategory.ProductCategory, ProductType.ProductType, Product.Product 
from ProductType 
right join Product 
on Product.ProductTypeID=ProductType.ProductTypeID 
left join ProductCategory 
on ProductType.ProductCategoryID=ProductCategory.ProductCategoryID;

3)use SourceSystem;
select DISTINCT ChannelCategory.ChannelCategory, Channel.Channel, 
CONCAT(Store.Address, ', ', Store.City, ', ', Store.StateProvince, ', ',Store.Country) 
from SalesHeader 
right join Store 
on SalesHeader.StoreID=Store.StoreID 
left join Channel 
on SalesHeader.ChannelID=Channel.ChannelID 
left join ChannelCategory 
on Channel.ChannelCategoryID=ChannelCategory.ChannelCategoryID
union
select DISTINCT ChannelCategory.ChannelCategory, Channel.Channel, 
concat (Customer.FirstName, ' ', Customer.LastName) 
from SalesHeader 
right join Customer 
on SalesHeader.CustomerID=Customer.CustomerID 
left join Channel 
on SalesHeader.ChannelID=Channel.ChannelID 
left join ChannelCategory 
on Channel.ChannelCategoryID=ChannelCategory.ChannelCategoryID
union
select DISTINCT ChannelCategory.ChannelCategory, Channel.Channel, Reseller.ResellerName
from SalesHeader 
right join Reseller 
on SalesHeader.ResellerID=Reseller.ResellerID 
left join Channel 
on SalesHeader.ChannelID=Channel.ChannelID 
left join ChannelCategory 
on Channel.ChannelCategoryID=ChannelCategory.ChannelCategoryID;
